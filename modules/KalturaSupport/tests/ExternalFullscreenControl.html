External Fullscreen Control Example

This example shows how to take control over the full screen functionality.
Changing full screen mode is done externally and the player is notified on the change.
To set it up:

  1. Set the global Flashvar 'EmbedPlayer.ExternalFullScreenControl' to true.
  2. Register to the player 'onToggleFullscreenButtonPress' event to enter and exit full screen using your ustom code.
  3. Notify the player on the new state by sending the 'onFullscreenStateChange' notification with the current state as a parameter.


Sample embed code: 
    var isFullScreen = false;

    function enterFullscreen(docElm){
        if (docElm.requestFullscreen) {
            docElm.requestFullscreen();
        }
        else if (docElm.mozRequestFullScreen) {
            docElm.mozRequestFullScreen();
        }
        else if (docElm.webkitRequestFullScreen) {
            docElm.webkitRequestFullScreen();
        }
        else if (docElm.msRequestFullscreen) {
            docElm.msRequestFullscreen();
        }
    }

    function exitFullscreen(){
        if (document.exitFullscreen) {
            document.exitFullscreen();
        }
        else if (document.mozCancelFullScreen) {
            document.mozCancelFullScreen();
        }
        else if (document.webkitCancelFullScreen) {
            document.webkitCancelFullScreen();
        }
        else if (document.msExitFullscreen) {
            document.msExitFullscreen();
        }
    }

    mw.setConfig('EmbedPlayer.ExternalFullScreenControl', true);

    kWidget.embed({
        'targetId': 'kaltura_player',
        'wid': '_243342',
        'uiconf_id' : '12905712',
        'entry_id' : '0_uka1msg4',
        'readyCallback': function(id){
            var kdp = document.getElementById(id);

            kdp.kBind("onToggleFullscreenButtonPress", function(){

                if (!isFullScreen){
                    enterFullscreen(kdp);
                }else{
                    exitFullscreen();
                }

                var fullscreenState = isFullScreen ? "exitFullscreen" : "enterFullscreen";
                kdp.sendNotification("onFullscreenStateChange", fullscreenState);

                isFullScreen = !isFullScreen;
            });
        }
    });
