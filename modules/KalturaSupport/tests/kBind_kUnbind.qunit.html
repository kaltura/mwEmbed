var foo = "bar";
// Unlike addJsListener, you can directly issue local callbacks and access parent scope.
kdp.kUnbind( "doPlay.test", function(){ //notice the .test postfix namespaces the event)
	console.log("play " + foo); // will log "play bar"
});
//Now we can unbind our .test postfix via kUnbind
kdp.kUbnind('.test')
