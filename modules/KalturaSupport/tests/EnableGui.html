Enable / Disable GUI

Enable GUI | Dissable GUI
Sample API code: 
kWidget.addReadyCallback(function( playerId ){
	var kdp = $('#' + playerId)[0];
	$('#enableGUI').click(function(){
		kdp.sendNotification('enableGui', { 'guiEnabled' : true } );
	});
	$('#disableGUI').click(function(){
		kdp.sendNotification('enableGui', { 'guiEnabled' : false } );
	});
});
